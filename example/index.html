KVStorage Node.js Demo

const https = require("https");

https.request("https://storage.conchbrain.club/demo/set",{
    headers: {
        "content-type": "application/json"
    },
    method: "PUT"
},(res) => {
    let data = "";

    res.on("data", (chunk) => {
        data += chunk;
    });

    res.on("end",() => {
        console.log(data);
    });
    
}).end(JSON.stringify({
    "key": "Hi",
    "value": "ConchBrain KVStorage"
}),(err) => {
    if(err)
        console.error(err);
});
            
const https = require("https");

https.get("https://storage.conchbrain.club/demo/get?Hi",(res) => {

    let data = "";

    res.on("data",(chunk) => {
        data += chunk;
    });

    res.on("end",() => {
        console.log("Hello " + JSON.parse(data));
    })
});
            
const https = require("https");

https.get("https://storage.conchbrain.club/demo/",(res) => {

    let data = "";

    res.on("data",(chunk) => {
        data += chunk;
    });

    res.on("end",() => {
        JSON.parse(data).forEach((element) => {
            console.log(element.value);
        });
    })
});
